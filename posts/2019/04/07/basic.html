如何开始一个taro项目

# taro 使用

# 1. init 项目初始化

npm i -g @taro/cli
taro init <project name>
cd <project name>
npm i taro-ui

# 2. run 初次启动项目

# npm script ./package.json
npm run dev:weapp
npm run build:weapp
# hotload
taro build --type weapp --watch

# 3. 安装taro ui

npm install taro-ui
npm install classnames --save

// 自定义主题颜色
$color-brand: #FFDD2B;
@import "~taro-ui/dist/style/index.scss";

# Tip

  1. 空格\t一次有效
  2. 路由传参
// preview.js
// 传
Taro.navigateTo({ // Taro.redirectTo
  url: `/pages/item/item?testId=11`
})
// 接
componentWillMount(){
    console.log(this.$router.params)
}
  1. 本地资源引用
//
import imgUrl from '../../assets/image/default.img'
<Image src={imgUrl}>
  1. 引用全局样式
export default ComponentName extends Component{
    static options = {
        addGlobalClass:true
    }
}
  1. 创建ref

// upfloor Component
class Father extends Component{
    roar(){
        this.cat.miao()
    }
    refCat = (node) => this.cat = node
    render(){
        return <Cat ref={this.refCat}/>
    }
}
// downfloor component
class Cat extends Component{
    miao(){
        console.log("我们一起学喵叫")
    }
    render(){
        return <View/>
    }
}
