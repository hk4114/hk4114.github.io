### 题目:输出字符串中出现最多字母

> var str = 'aassddddfffff'

#### 方法1

```js

function theMost(str){
    return str.split('').sort()          // ["a", "a", "d", "d", "d", "d", "f", "f", "f", "f", "f", "s", "s"]
    .join('').match(/(\S)\1*/g)          // ["aa", "dddd", "fffff", "ss"]
    .sort((a,b)=>{
            return b.length-a.length    // 按照字符串长度排序
     })[0][0]

}

console.log(theMost(str))

```
